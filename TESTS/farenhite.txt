;; 1. Add Converter to Gateway of Tally
[#Menu: Gateway of Tally]
    Add : Item : After : Quit : "Fahrenheit Converter" : Alter : CTEMP_FahrToCelReport

;; 2. Report Definition
[Report: CTEMP_FahrToCelReport]
    Form : CTEMP_FahrToCelForm
    Variable : CTEMP_FahrVar
    Set : CTEMP_FahrVar : 32

;; 3. Form Definition 
[Form: CTEMP_FahrToCelForm]
    Part : CTEMP_HdrPart, CTEMP_InPart, CTEMP_OutPart
    Width : 35% Screen
    Height : 25% Screen
    Bottom Button : CTEMP_ClearBtn

;; 4. Parts
[Part: CTEMP_HdrPart]
    Lines : CTEMP_HdrLine

[Part: CTEMP_InPart]
    Lines : CTEMP_InLine

[Part: CTEMP_OutPart]
    Lines : CTEMP_OutLine

;; 5. Lines
[Line: CTEMP_HdrLine]
    Fields : CTEMP_HdrFld

[Line: CTEMP_InLine]
    Fields : CTEMP_LblFld, CTEMP_FahrFld

[Line: CTEMP_OutLine]
    Fields : CTEMP_ResLblFld, CTEMP_CelFld

;; 6. Fields
[Field: CTEMP_HdrFld]
    Use : Name Field
    Set As : "Fahrenheit to Celsius"
    Style : Large
    Full Width : Yes
    Align : Center

[Field: CTEMP_LblFld]
    Use : Medium Prompt
    Set As : "Fahrenheit :"
    Align : Right

[Field: CTEMP_FahrFld]
    Use : Number Field
    Variable : CTEMP_FahrVar
    Align : Left

[Field: CTEMP_ResLblFld]
    Use : Medium Prompt
    Set As : "Celsius :"
    Align : Right

[Field: CTEMP_CelFld]
    Use : Number Field
    ;; Using / 1.8 prevents integer math parsing errors in Tally
    Set As : (#CTEMP_FahrFld - 32) / 1.8
    Set Always : Yes
    Skip : Yes
    Align : Left
    Style : Bold
    Color : Dark Green

;; 7. Variable Declarations
[Variable: CTEMP_FahrVar]
    Type : Number
    Volatile : Yes

;; 8. Clear Button & Function
[Button: CTEMP_ClearBtn]
    Key : Alt + C
    Title : "Clear"
    Action : Call : CTEMP_ClearFunc

[Function: CTEMP_ClearFunc]
    10 : Set : CTEMP_FahrVar : 32
