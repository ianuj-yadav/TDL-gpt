[System: Formula]
	DestPath			: $$LocaleString:##DestinationPath +"\" + @@DateForm + "\" + @@TimeForm 
	DestDatePath		: $$StringPart:@@DestPath:0:(($$StringLength:@@DestPath)-7)
	ZipFileName1		: $$StringPart:@@DestPath:(($$StringLength:@@DestPath)-6):($$StringLength:@@DestPath);@@DestPath+".zip"
	ZipFileName			: @@DestPath+".zip";@@ZipFileName1+".zip"
	CoNumber			: $$Number:($$String:$CompanyNumber:Company:$name);$$String:($$Number:($CompanyNumber:Company:##SVCurrentCompany))
	CNumber				: $CompanyNumber:Company:$name
	DateForm			: $$LocaleString:$$MachineDate+"-v"+@@Releaseversion
	TimeForm			: @@HrsForm + @@MtsForm + @@SesForm
	HrsForm				: $$StringPart:($$LocaleString:($$SysInfo:SystemTimeHMS)):0:2
	MtsForm				: $$StringPart:($$LocaleString:($$SysInfo:SystemTimeHMS)):3:2
	SesForm				: $$StringPart:($$LocaleString:($$SysInfo:SystemTimeHMS)):6:2
	

	Release  			: "%s"
	Releaseversion		: $$SPrintf:@@Release:@@VersionReleaseString
	PopMessage			: "https://tallyatcloud.com/tallybackup/"+@@SrNo + @@EmailId + @@UserType 
	SrNo				: $$LocaleString:($$LicenseInfo:SerialNumber);+"/" 
	EmailId				: $$LocaleString:($$LicenseInfo:AdminEmailID);+"/" 
	UserType			: If $$LicenseInfo:IsSilver then "SILVER" Else If $$LicenseInfo:IsGold Then "GOLD" Else "SERVER"
	TSSExpiry			: $$LocaleString:($$Date:$$TNSubscValidTillInDateFormat)
	TotalUrl			: @@SrNo+"*"+@@EmailId+"*"+@@TSSExpiry+"*"+@@UserType
	EncryptionPassword	: @@SrNo+"12345678"
	EncUrl				: $$EncryptStr:@@TotalUrl:@@EncryptionPassword
	NewEncUrl			: $$UrlModifier:@@EncUrl
	;FinalEncUrlwithPass	: @@EncUrl+"/"+@@EncryptionPassword
