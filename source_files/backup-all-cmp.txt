[#Menu:gateway of tally]
	Add:Button:Call Function

[Button : Call Function]
	Key : Alt + F9
	Title : "Backup"
	Action		: Call : TestFunction
	
[Function: TestFunction]
	Variable	: BackupDetVar	: String
	00	: IF	: ##EnableAutoBackup
	10 	: IF	: ##AutoBackupPrompt
	20 	: QUERYBOX : "Backup Company ?" :Yes:No
	30	: IF 	: $$LastResult
	40	: CALL	: Backup Function
	50	: ENDIF
	60	: ELSE	: 
	70	: CALL	: Backup Function 
	80 	: ENDIF
	90	: ENDIF
	;100	: Log	: "Done"

[Function: Backup Function]
	Variable	: Cmps		: String
	Variable	: BackupDetVar	: String
	10	: Log	: "Checking"
	100	: Set	: Cmps		: ""
	110	: Set	: BackupDetVar	:""
	200	: Walk Collection	: List of Companies On Disk
	250	: Set		: Cmps	: ""
	300	: Set		: Cmps	: @@DestPath + "," +  ##SVCurrentPath +","+ $name + "," +$$CommaList:$CompanyNumber+","
	400   : SET	: BackupDetVar	: ##BackupDetVar+##Cmps
	600	: End Walk
	1000	: Set	: BackupDetVar	: $$StringPart:##BackupDetVar:0:(($$StringLength:##BackupDetVar)-1)
	1200	: BACKUP COMPANY: "," : ##BackupDetVar
	;1300	: Log		: ##BackupDetVar

	

[System: Formula]
	DestPath	: $$LocaleString:##DestinationPath +"\" + @@DateForm + "\" + @@TimeForm 
	CoNumber	: $$Number:($$String:$CompanyNumber:Company:$name);$$String:($$Number:($CompanyNumber:Company:##SVCurrentCompany))
	CNumber		: $CompanyNumber:Company:$name
	DateForm	: $$LocaleString:$$MachineDate
	TimeForm	: @@HrsForm + @@MtsForm + @@SesForm
	HrsForm		: $$StringPart:($$LocaleString:($$SysInfo:SystemTimeHMS)):0:2
	MtsForm		: $$StringPart:($$LocaleString:($$SysInfo:SystemTimeHMS)):3:2
	SesForm		: $$StringPart:($$LocaleString:($$SysInfo:SystemTimeHMS)):6:2
