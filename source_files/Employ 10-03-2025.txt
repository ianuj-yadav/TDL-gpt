;; ====================================================
;; ? CREATE A NEW OBJECT CALLED "Employee"
;; ====================================================
[Collection: EmployeeCollection]
    Type    : Employee
    Childof : "Employees"
    Title   : "Employee List"
    Fetch   : Name,  Address, DateOfJoining, Position, Gender

[Object: Employee]
    Name        : String
   ; Age         : String
    Address     : String
    DateOfJoining : Date
    Position    : String
    Gender      : String
    IsSaved     : Logical

[System:UDF]
	Name        : String	: 12001
    ;Age         : String	: 12002
    Address     : String	: 12003
    DateOfJoining : Date	: 12004
    Position    : String	: 12005
    Gender      : String	: 12006
    IsSaved     : Logical	: 12007
;; ====================================================
;; ? CREATE A FORM TO ENTER EMPLOYEE DATA
;; ====================================================
[Report: EmployeeForm]
    Form : EmployeeForm
    Title : "Add New Employee"

[Form: EmployeeForm]
    Part    : EmployeeFormPart
    Button  : SaveEmployeeButton
    Width   : 50% Page
    Height  : 60% Page
    Space Bottom : 2

[Part: EmployeeFormPart]
    Line: EmployeeNameLine,  EmployeeAddressLine, EmployeeDOJLine, EmployeePositionLine, EmployeeGenderLine

[Line: EmployeeNameLine]
    Field: Medium Prompt, EmployeeName1
	Local	: Field	: MediumPrompt	: Set as: "Employee Name:"
    Local	: Field	: MediumPrompt	: Width  : 20   
[Field: EmployeeName1]
    Type : String
    Storage : Name
    Width  : 30
    Set Always : Yes



[Line: EmployeeAddressLine]
    Field: Medium Prompt, EmployeeAddress1
	Local	: Field	: MediumPrompt	: Set as: "Address:"
    Local	: Field	: MediumPrompt	: Width  : 20 
[Field: EmployeeAddress1]
    Type : String
    Storage : Address
    Width  : 30
    Set Always : Yes

[Line: EmployeeDOJLine]
    Field: Medium Prompt, EmployeeDOJ1
	Local	: Field	: MediumPrompt	: Set as: "Date of Joining:"
    
    Local	: Field	: MediumPrompt	: Width  : 20 
[Field: EmployeeDOJ1]
    Type : Date
    Storage : DateOfJoining
    Width  : 30
    Set Always : Yes

[Line: EmployeePositionLine]
    Field: Medium Prompt, EmployeePosition1
Local	: Field	: MediumPrompt	: Set as: "Position:"
    
    Local	: Field	: MediumPrompt	: Width  : 20
[Field: EmployeePosition1]
    Type : String
    Storage : Position
    Width  : 30
    Set Always : Yes

[Line: EmployeeGenderLine]
    Field: Medium Prompt, EmployeeGender1
	Local	: Field	: MediumPrompt	: Set as: "Gender:"
    Local	: Field	: MediumPrompt	: Width  : 20

[Field: EmployeeGender1]
    Type : String
    Storage : Gender
    Width  : 30
    Set Always : Yes

;; ====================================================
;; ? SAVE BUTTON TO STORE EMPLOYEE DATA
;; ====================================================
[Button: SaveEmployeeButton]
    Title : "Save Employee"
    Key : Enter
    Action : Create:Employee

[Function: Create:Employee]
    Variable: Success : Logical

    00 : Create Target : Employee
    10 : Set Value : Name         : ##Name
    ;20 : Set Value : Age          : ##Age
    30 : Set Value : Address      : ##Address
    40 : Set Value : DateOfJoining: ##DateOfJoining
    50 : Set Value : Position     : ##Position
    60 : Set Value : Gender       : ##Gender
    70 : Set Value : IsSaved      : Yes
    80 : Set Target

    90 : MSGBOX : "Success" : "Employee Data Saved Successfully!"

    ;; Automatically Close the Form after Save
    100 : RETURN

;; ====================================================
;; ? CREATE MENU IN GATEWAY OF TALLY
;; ====================================================
[#Menu: Gateway of Tally]
    Add: Item: "Employee Master" : Alter : EmployeeForm

;; ====================================================
;; ? VIEW EMPLOYEE DATA LATER
;; ====================================================
[Report: EmployeeList]
    Form : EmployeeListForm
    Title: "Employee List"

[Form: EmployeeListForm]
    Part : EmployeeListPart

[Part: EmployeeListPart]
    Line : EmployeeHeader, EmployeeDataLine

    Repeat : EmployeeDataLine : EmployeeCollection
    Scroll : Vertical

[Line: EmployeeHeader]
    Field : Medium Prompt,  Medium Prompt, Medium Prompt
    Local : Field : Medium Prompt : Set As : "Name"
    ;Local : Field : Medium Prompt : Set As : "Age"
    Local : Field : Medium Prompt : Set As : "Position"
    Local : Field : Medium Prompt : Set As : "Date Of Joining"

[Line: EmployeeDataLine]
    Field : EmployeeNameList,  EmployeePositionList, EmployeeDOJList

[Field: EmployeeNameList]
    Set As : $Name
    Width  : 20


[Field: EmployeePositionList]
    Set As : $Position
    Width  : 20

[Field: EmployeeDOJList]
    Set As : $$String:$DateOfJoining
    Width  : 20

;; ====================================================
;; ? ADD MENU TO VIEW EMPLOYEE DATA
;; ====================================================
[#Menu: Gateway of Tally]
    Add: Item: "View Employee Data" : Display : EmployeeList
