
[#Menu:gateway of tally]
	Add		: Button		: Call Function
	Add		: Item	: "All Companies"	: Display	: AllCompanies



[Report:AllCompanies]
	Form		: AllCompanies
	
[Form:AllCompanies]
	Part		: AllCompanies
	
[Part:AllCompanies]
	Line		: AllCompanies
	Repeat		: AllCompanies		: List of Companies On Disk;List of Primary Companies
	Scroll		: Vertical
	
	[Line:AllCompanies]
		Fields		: AllCompanies, CmpNo,DateTime1
		
		[Field:AllCompanies]
			Use			: Name Field
			Set as		: $name
			Line		: 0
			Width		: 40% Page
			
		[Field:CmpNo]
			Use			: Name Field
			Set as		: $$LocaleString:@@DestPath;$$Number:($$String:$CompanyNumber:Company:$name)
			Line		: 0
			Width		: 50% Page
			
		[Field:DateTime1]
			Use			: Name Field
			Set as		: $$StringPart:("000"+$$String:$CompanyNumber):(($$StringLength:@number)-6):6;$$CommaList:$CompanyNumber;$$ReadINI:@path1:"TALLY":"Default Companies"
			Line		: 0
			Width		: 10% Page
			path1		: $$SysInfo:ApplicationPath + "\tally.ini"
			time		: $$SysInfo:SystemTimeHMS
			number		: "000"+$$String:$CompanyNumber
			Set Always	: yes
